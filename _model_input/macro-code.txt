

Option Explicit

Sub save_tabs_to_csv()
    Dim csv_filename As String
    Dim CurrentWB As Workbook, TempWB As Workbook
    Dim WS_Count As Integer
    Dim I As Integer
    Dim current_sheet As Worksheet
    
    
    ' count worksheets
    WS_Count = ActiveWorkbook.Worksheets.Count
    
    Set CurrentWB = ActiveWorkbook
    
    For I = 1 To WS_Count
        ' MsgBox ActiveWorkbook.Worksheets(I).Name
        
        Set current_sheet = ActiveWorkbook.Worksheets(I)
        current_sheet.UsedRange.Copy
    
        Set TempWB = Application.Workbooks.Add(1)
        With TempWB.Sheets(1).Range("A1")
          .PasteSpecial xlPasteValues
          ' .PasteSpecial xlPasteFormats
        End With
    
        csv_filename = "C:\Users\vdesai\Git\bootstrapper-og\_model_input\_csv_drivers\" & current_sheet.Name & ".csv"
    
        Application.DisplayAlerts = False
        TempWB.SaveAs Filename:=csv_filename, FileFormat:=xlCSV, CreateBackup:=False, Local:=True
        TempWB.Close SaveChanges:=False
        Application.DisplayAlerts = True
    Next I
End Sub
